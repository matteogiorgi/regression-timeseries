import pandas as pd
import statsmodels.api as sm

# ho messo i dati dentro in csv come vuole la prof.
df = pd.read_csv("AmesHousing.csv")

# mi salvo la variabile target e i regressori
y = df["SalePrice"]
regressors = [
    "Gr Liv Area",
    "Overall Qual",
    "Overall Cond",
    "Year Built",
    "Total Bsmt SF",
    "Garage Cars",
    "Garage Area",
    "Lot Area",
    "House Style",
    "Neighborhood",
    "Central Air",
    "Kitchen Qual",
]

# seleziono solo le colonne di interesse
# e mi creo le dummies per le categoriche
X = df[regressors]
X = pd.get_dummies(X, drop_first=True)

# converto tutto in numerico
# (ci possono essere cose strane -> NaN)
X = X.apply(pd.to_numeric, errors="coerce")
y = pd.to_numeric(y, errors="coerce")

# metto insieme y e X
# tolgo tutte le righe con NaN
data = pd.concat([y, X], axis=1).dropna()

# separo di nuovo y e X ma con NumPy
y_clean = data["SalePrice"].to_numpy(dtype=float)
X_clean = data.drop(columns=["SalePrice"]).to_numpy(dtype=float)

# aggiungo l'intercetta sugli array NumPy
X_clean = sm.add_constant(X_clean)

# regressione OLS
model = sm.OLS(y_clean, X_clean).fit()
print(model.summary())
